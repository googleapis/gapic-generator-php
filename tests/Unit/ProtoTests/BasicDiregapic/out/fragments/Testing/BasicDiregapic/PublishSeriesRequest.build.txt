
    /**
     * @param \Testing\BasicDiregapic\ShelfResponse      $shelf      The shelf in which the series is created.
     * @param \Testing\BasicDiregapic\BookResponse[]     $books      The books to publish in the series.
     * @param int                                        $edition    The edition of the series
     * @param \Testing\BasicDiregapic\SeriesUuidResponse $seriesUuid Uniquely identifies the series to the publishing house.
     * @param string                                     $publisher  The publisher of the series.
     *
     * @return \Testing\BasicDiregapic\PublishSeriesRequest
     *
     * @experimental
     */
    public static function build(\Testing\BasicDiregapic\ShelfResponse $shelf, array $books, int $edition, \Testing\BasicDiregapic\SeriesUuidResponse $seriesUuid, string $publisher): self
    {
        return (new self())
            ->setShelf($shelf)
            ->setBooks($books)
            ->setEdition($edition)
            ->setSeriesUuid($seriesUuid)
            ->setPublisher($publisher);
    }
